$.getJSON("https://api.macaumarksix.com/api/macaujc.com", function(data) {

var wave = result.wave.split(",");
var html = "<h2>老奥</h2>" +
"<p>期号：" + result.expect + "</p>" +
"<p>号码：";
for (var i = 0; i < openCode.length; i++) {
html += "<span style='color: " + wave[i] + "'>" + openCode[i] + "</span>";
if (i < openCode.length - 1) {
html += ", ";
}
}
html += "</p>" +
"<p>属相：" + result.zodiac + "</p>" +
"<p>更新时间：" + result.openTime + "</p>";
// "<p>颜色：" + result.wave + "</p>" +
// "<p>信息：" + result.info + "</p>";
$("#api1").html(html);
});
